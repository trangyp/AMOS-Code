## 🎮 Usage

### Quick Start
```python
from amos_brain.knowledge_loader import (
    load_brain_knowledge,
    query_knowledge,
    get_knowledge_brain
)

# Load knowledge
result = load_brain_knowledge()
print(f"Loaded: {result['knowledge_stats']['total_entries']} entries")

# Query knowledge
results = query_knowledge("architecture", limit=5)
for r in results:
    print(f"- {r['key']} (domain: {r['domain']})")
```

### Advanced Usage
```python
from amos_brain.knowledge_loader import KnowledgeEnhancedBrain

# Create brain with knowledge
brain = KnowledgeEnhancedBrain()
brain.initialize()

# Query with domain filter
entries = brain.knowledge_loader.query(
    "scalability",
    domain="software",
    limit=3
)

# Think with knowledge enhancement
result = brain.think_with_knowledge(
    "What architecture should I use?",
    context={"constraints": ["high_availability"]}
)
print(f"Used {result['knowledge_used']} knowledge entries")
```
